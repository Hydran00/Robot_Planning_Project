230bool getObstacles(std::string const & filename, std::vector<Obstacle> & map)
231{
232 map.clear();
233
234 std::ifstream input(filename);
235 if (!input.is_open())
236 {
237 std::cerr << "Error opening file " << filename << std::endl;
238 return false;
239 }
240
241 try
242 {
243 std::string str((std::istreambuf_iterator<char>(input)), std::istreambuf_iterator<char>());
244 nlohmann::json jdata = nlohmann::json::parse(str);
245 return getObstacles(jdata, map);
246 }
247 catch (...)
248 {}
249
250 input.clear(); // clear fail and eof bits
251 input.seekg(0, std::ios::beg); // back to the start!
252
253 const std::string PARSE_ERROR_MSG = "Error while parsing map file";
254
255 while (!input.eof())
256 {
257 int size;
258 if (!(input >> size))
259 {
260 if (input.eof())
261 break;
262 std::cerr << PARSE_ERROR_MSG << std::endl;
263 return false;
264 }
265
266 Obstacle curr;
267 for (int i=0; i<size; ++i)
268 {
269 double x, y;
270 if (!(input >> x >> y))
271 {
272 std::cerr << PARSE_ERROR_MSG << std::endl;
273 return false;
274 }
275
276 curr.push_back({x, y});
277 }
278
279 if (curr.back()!=curr.front())
280 {
281 curr.push_back(curr.front());
282 }
283
284 map.push_back(curr);
285 }
286
287 return true;
