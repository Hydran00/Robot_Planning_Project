75// Receives all message parts from socket, prints neatly
76//
77static inline void
78s_dump (zmq::socket_t & socket)
79{
80 std::cout << "----------------------------------------" << std::endl;
81
82 while (1) {
83 // Process all parts of the message
84 zmq::message_t message;
85 socket.recv(&message);
86
87 // Dump the message as text or binary
88 int size = message.size();
89 std::string data(static_cast<char*>(message.data()), size);
90
91 bool is_text = true;
92
93 int char_nbr;
94 unsigned char byte;
95 for (char_nbr = 0; char_nbr < size; char_nbr++) {
96 byte = data [char_nbr];
97 if (byte < 32 || byte > 127)
98 is_text = false;
99 }
100 std::cout << "[" << std::setfill('0') << std::setw(3) << size << "]";
101 for (char_nbr = 0; char_nbr < size; char_nbr++) {
102 if (is_text)
103 std::cout << (char)data [char_nbr];
104 else
105 std::cout << std::setfill('0') << std::setw(2)
106 << std::hex << (unsigned int) data [char_nbr];
107 }
108 std::cout << std::endl;
109
110 int more = 0; // Multipart detection
111 size_t more_size = sizeof (more);
112 socket.getsockopt (ZMQ_RCVMORE, &more, &more_size);
113 if (!more)
114 break; // Last message part
115 }
116}

171// Return current system clock as milliseconds
172static inline int64_t
173s_clock (void)
174{
175#if (defined (WIN32))
176 FILETIME fileTime;
177 GetSystemTimeAsFileTime(&fileTime);
178 unsigned __int64 largeInt = fileTime.dwHighDateTime;
179 largeInt <<= 32;
180 largeInt |= fileTime.dwLowDateTime;
181 largeInt /= 10000; // FILETIME is in units of 100 nanoseconds
182 return (int64_t)largeInt;
183#else
184 struct timeval tv;
185 gettimeofday (&tv, NULL);
186 return (int64_t) (tv.tv_sec * 1000 + tv.tv_usec / 1000);
187#endif
188}
